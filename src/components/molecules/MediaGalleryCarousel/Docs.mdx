## Example Usage(s)
```tsx
<MediaGalleryCarousel
  {...{
    slides: [
        {
            image: (
                <JSSImage
                    field={{
                        src: '/images/hero-banners/advocatehealth-merger-kids.webp'
                    }}
                />
            ),
            headline: <JSSText field={{
                value: 'Title of image. Lorem ipsum dolor sit amet.'
            }} />,
            supportCopy: <JSSRichText field={{
                value: '<p>Image description and <a href="#">link</a> goes here.</p>'
            }} />
        },
        {
            image: (
                <JSSImage
                    field={{
                        src: 'https://www.advocatechildrenshospital.com/_widgets/banners/easset_upload_file71272_456927_e.webp'
                    }}
                />
            )
        },
        {
            videoId: 'kjmrWe8rsAY'
        }
    ],
    shouldHideNavigation: false,
    shouldHideThumbnails: false
}}
/>
```
